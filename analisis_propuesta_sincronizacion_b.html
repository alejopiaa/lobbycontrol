2. Flujo Detallado del Sistema

📤 Fase A: El Administrador Actualiza y Publica

01

Sincronización de Excel

El Administrador importa el archivo lobby_data.xlsx. El backend procesa los registros y los graba en su lobby.db local.

02

Generación de Metadatos

Se genera un archivo ligero version.json que contiene la marca de tiempo de la actualización, el tamaño del archivo y el hash MD5.

03

Copia a Carpeta Local Sync

La aplicación copia lobby.db y version.json al directorio de OneDrive configurado localmente en Windows.

04

Sincronización OneDrive

El cliente nativo de OneDrive en Windows detecta los archivos modificados y los sube de manera inmediata a la nube.

📥 Fase B: El Usuario Común Inicia la Aplicación

01

Verificación de Versión

Al iniciar la aplicación, se realiza una consulta rápida (HTTP GET) al archivo version.json remoto para verificar si hay datos nuevos.

02

Comparación Remota vs Local

Se contrasta la marca de tiempo (o hash) del archivo descargado con el valor almacenado en la tabla de configuración local.

03

Descarga en Temporal

Si hay una nueva versión, se descarga la base de datos directamente al archivo temporal data/lobby.db.tmp.

04

Cierre, Reemplazo y Re-apertura

Se cierran de forma segura todas las conexiones SQLite, se sustituye el archivo lobby.db y se vuelve a conectar la BD en Express de forma transparente.

src/config/db-sync.js

Nueva utilidad en los clientes que se encarga de descargar la base de datos de forma segura al abrir la aplicación.

const fs = require('fs');
const path = require('path');
const https = require('https');

function downloadFile(url, destPath) {
  return new Promise((resolve, reject) => {
    const file = fs.createWriteStream(destPath);
    https.get(url, (response) => {
      if (response.statusCode !== 200) {
        return reject(new Error(`Error de descarga: Código ${response.statusCode}`));
      }
      response.pipe(file);
      file.on('finish', () => {
        file.close(resolve);
      });
    }).on('error', (err) => {
      fs.unlinkSync(destPath);
      reject(err);
    });
  });
}

async function checkAndSyncDatabase(dbConnectionManager) {
  const versionUrl = process.env.REMOTE_VERSION_URL;
  const dbUrl = process.env.REMOTE_DB_URL;
  
  if (!versionUrl || !dbUrl) return;

  const localVersionPath = path.join(__dirname, '../../data/version.json');
  const localDbPath = path.join(__dirname, '../../data/lobby.db');
  const tempDbPath = path.join(__dirname, '../../data/lobby.db.tmp');
  const tempVersionPath = path.join(__dirname, '../../data/version.json.tmp');

  try {
    await downloadFile(versionUrl, tempVersionPath);
    const remoteVersion = JSON.parse(fs.readFileSync(tempVersionPath, 'utf8'));
    
    let localVersion = { last_import_timestamp: 'Nunca' };
    if (fs.existsSync(localVersionPath)) {
      localVersion = JSON.parse(fs.readFileSync(localVersionPath, 'utf8'));
    }

    if (remoteVersion.last_import_timestamp !== localVersion.last_import_timestamp) {
      console.log('Nueva versión disponible. Iniciando actualización...');
      await downloadFile(dbUrl, tempDbPath);
      
      // Bloque de Reemplazo Seguro
      await dbConnectionManager.closeConnection();
      fs.copyFileSync(tempDbPath, localDbPath);
      fs.copyFileSync(tempVersionPath, localVersionPath);
      
      fs.unlinkSync(tempDbPath);
      fs.unlinkSync(tempVersionPath);
      await dbConnectionManager.openConnection();
      console.log('✓ Base de datos sincronizada correctamente.');
    } else {
      console.log('Base de datos al día.');
      fs.unlinkSync(tempVersionPath);
    }
  } catch (err) {
    console.error('Fallo en sincronización remota:', err.message);
  }
}
